Setup a server using express 

const express = require("express");
const app = express()
const port = 4000
app.get('/',(req,res,next)=>{
res.send("hello world")
})
app.listen(port,()=>{
console.log(app is running on port ${4000})
})
//  node app.js to execute
// output - hello world

// this is app .js 
const express = require("express");
const bodyparser= require('body-parser');


const app = express();


const adminroutes= require('./Routes/admin');
const shoproutes= require("./Routes/shop")


 
 const port = 4000


 app.use(bodyparser.urlencoded({extended:false}));

 app.use(adminroutes);
 app.use(shoproutes); 
 



 app.listen(port,()=>{
 console.log(`app is running on port ${4000}`)
 })

Filter Routes in express

Bassically if you want to nested route like this http://localhost:4000/admin/add-product then simply put
app.use(" /admin",adminroutes); in app.js file . 

 //this is app js 
 const express = require("express");
const bodyparser= require('body-parser');


const app = express();


const adminroutes= require('./Routes/admin');
const shoproutes= require("./Routes/shop")


 
 const port = 4000


 app.use(bodyparser.urlencoded({extended:false}));

 app.use(" /admin",adminroutes);
 app.use(shoproutes); 
 



 app.listen(port,()=>{
 console.log(`app is running on port ${4000}`)
 })

const { Router } = require('express');
const express =require('express')
  

const router = express.Router();

// /admin/add-product ==>GET
router.get ('/add-product',(req,res,next)=>{
    res.send('<form action ="/admin/add-product" method ="POST"><input type ="text" name ="title"><button type ="submit" >add product</button></form>')
})

//    /admin/add-product ==>POST 
router .post('/add-product',(req,res,next)=>{
   console.log(req.body);
   res.redirect('/')
}) 



module.exports = router;
